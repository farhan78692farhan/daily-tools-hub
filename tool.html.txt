        // QR کوڈ جنریٹر فنکشنز
        function generateQR() {
            const text = document.getElementById('qrText').value;
            if (!text) {
                alert('براہ کرم کوئی ٹیکسٹ یا URL درج کریں');
                return;
            }
            
            const qrCodeDiv = document.getElementById('qrCode');
            qrCodeDiv.innerHTML = '';
            
            // Using a simple QR code API (you might want to use a library for offline use)
            const qrImg = document.createElement('img');
            qrImg.src = `https://api.qrserver.com/v1/create-qr-code/?size=150x150&data=${encodeURIComponent(text)}`;
            qrImg.alt = 'QR Code';
            
            qrCodeDiv.appendChild(qrImg);
            document.getElementById('downloadBtn').style.display = 'inline-block';
        }
        
        function downloadQR() {
            const qrImg = document.querySelector('#qrCode img');
            if (!qrImg) return;
            
            const link = document.createElement('a');
            link.href = qrImg.src;
            link.download = 'qrcode.png';
            link.click();
        }

        // کرنسی کنورٹر فنکشنز
        async function convertCurrency() {
            const from = document.getElementById('currencyFrom').value;
            const to = document.getElementById('currencyTo').value;
            const amount = document.getElementById('currencyAmount').value;
            
            if (from === to) {
                document.getElementById('currencyResult').textContent = amount;
                return;
            }
            
            try {
                // Note: For a real application, you should use a proper API with an API key
                // This is a simplified example that might not work due to CORS restrictions
                const response = await fetch(`https://api.exchangerate-api.com/v4/latest/${from}`);
                const data = await response.json();
                
                if (data && data.rates && data.rates[to]) {
                    const rate = data.rates[to];
                    const result = (amount * rate).toFixed(2);
                    document.getElementById('currencyResult').textContent = result;
                    document.getElementById('currencyDate').textContent = `تازہ ترین شرح: ${new Date(data.date).toLocaleDateString()}`;
                } else {
                    throw new Error('Invalid response');
                }
            } catch (error) {
                console.error('Error fetching exchange rates:', error);
                // Fallback rates (for demo purposes only)
                const fallbackRates = {
                    USD: { PKR: 278.50, EUR: 0.85, GBP: 0.73, JPY: 110.25, SAR: 3.75, AED: 3.67 },
                    EUR: { USD: 1.18, PKR: 328.75, GBP: 0.86, JPY: 130.50, SAR: 4.42, AED: 4.33 },
                    GBP: { USD: 1.37, EUR: 1.16, PKR: 381.20, JPY: 151.75, SAR: 5.14, AED: 5.03 },
                    PKR: { USD: 0.0036, EUR: 0.0030, GBP: 0.0026, JPY: 0.40, SAR: 0.013, AED: 0.013 },
                    SAR: { USD: 0.27, EUR: 0.23, GBP: 0.19, PKR: 74.33, JPY: 29.40, AED: 0.98 },
                    AED: { USD: 0.27, EUR: 0.23, GBP: 0.20, PKR: 75.85, JPY: 30.05, SAR: 1.02 }
                };
                
                if (fallbackRates[from] && fallbackRates[from][to]) {
                    const rate = fallbackRates[from][to];
                    const result = (amount * rate).toFixed(2);
                    document.getElementById('currencyResult').textContent = result + ' (تازہ ترین ڈیٹا دستیاب نہیں، پرانا ڈیٹا استعمال کیا گیا)';
                    document.getElementById('currencyDate').textContent = '';
                } else {
                    document.getElementById('currencyResult').textContent = 'تبادلہ شرح دستیاب نہیں';
                    document.getElementById('currencyDate').textContent = '';
                }
            }
        }

        // بیس64 انکوڈر/ڈیکوڈر فنکشنز
        function encodeBase64() {
            const text = document.getElementById('base64Input').value;
            document.getElementById('base64Output').value = btoa(unescape(encodeURIComponent(text)));
        }
        
        function decodeBase64() {
            const text = document.getElementById('base64Input').value;
            try {
                document.getElementById('base64Output').value = decodeURIComponent(escape(atob(text)));
            } catch (e) {
                document.getElementById('base64Output').value = 'غلط فارمیٹ! صرف درست بیس64 ٹیکسٹ ڈیکوڈ کیا جا سکتا ہے';
            }
        }
        
        function copyBase64() {
            const output = document.getElementById('base64Output');
            output.select();
            document.execCommand('copy');
            alert('کاپی ہو گیا!');
        }

        // ایم ڈی5 جنریٹر فنکشنز
        function generateMD5() {
            const text = document.getElementById('md5Input').value;
            if (!text) {
                alert('براہ کرم کوئی ٹیکسٹ درج کریں');
                return;
            }
            
            // Simple MD5 implementation (for demo purposes)
            // Note: For production, use a proper library like crypto-js
            document.getElementById('md5Output').value = simpleMD5(text);
        }
        
        function simpleMD5(input) {
            // This is a very simplified version for demonstration
            // A real MD5 algorithm is much more complex
            let hash = 0;
            for (let i = 0; i < input.length; i++) {
                const char = input.charCodeAt(i);
                hash = (hash << 5) - hash + char;
                hash |= 0; // Convert to 32bit integer
            }
            return (hash >>> 0).toString(16); // Convert to unsigned and then to hex
        }
        
        function copyMD5() {
            const output = document.getElementById('md5Output');
            output.select();
            document.execCommand('copy');
            alert('کاپی ہو گیا!');
        }

        // ٹو ڈو لسٹ فنکشنز
        function addTodo() {
            const input = document.getElementById('todoInput');
            const task = input.value.trim();
            
            if (!task) return;
            
            const todoList = document.getElementById('todoList');
            const li = document.createElement('li');
            
            const checkbox = document.createElement('input');
            checkbox.type = 'checkbox';
            checkbox.onchange = function() {
                if (this.checked) {
                    taskSpan.style.textDecoration = 'line-through';
                    taskSpan.style.color = '#999';
                } else {
                    taskSpan.style.textDecoration = 'none';
                    taskSpan.style.color = '#333';
                }
            };
            
            const taskSpan = document.createElement('span');
            taskSpan.textContent = task;
            taskSpan.style.marginLeft = '10px';
            
            const deleteBtn = document.createElement('button');
            deleteBtn.textContent = 'حذف کریں';
            deleteBtn.style.marginLeft = '10px';
            deleteBtn.style.padding = '2px 5px';
            deleteBtn.style.fontSize = '0.8em';
            deleteBtn.onclick = function() {
                li.remove();
            };
            
            li.appendChild(checkbox);
            li.appendChild(taskSpan);
            li.appendChild(deleteBtn);
            todoList.appendChild(li);
            
            input.value = '';
        }

        // رینڈم نمبر جنریٹر فنکشنز
        function generateRandom() {
            const min = parseInt(document.getElementById('randomMin').value) || 0;
            const max = parseInt(document.getElementById('randomMax').value) || 100;
            const count = parseInt(document.getElementById('randomCount').value) || 1;
            
            if (min >= max) {
                alert('زیادہ سے زیادہ قیمت کم از کم قیمت سے زیادہ ہونی چاہیے');
                return;
            }
            
            let numbers = [];
            for (let i = 0; i < count; i++) {
                numbers.push(Math.floor(Math.random() * (max - min + 1)) + min);
            }
            
            document.getElementById('randomResult').textContent = numbers.join(', ');
        }

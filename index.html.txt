<script>
  // ─────────────── ٹول 1 کے فنکشنز ───────────────
  function toUpper(){
    let text = document.getElementById('textInput').value;
    document.getElementById('textInput').value = text.toUpperCase();
    document.getElementById('textResult').innerText = 'متن بڑے حروف میں کر دیا گیا';
  }

  function toLower(){
    let text = document.getElementById('textInput').value;
    document.getElementById('textInput').value = text.toLowerCase();
    document.getElementById('textResult').innerText = 'متن چھوٹے حروف میں کر دیا گیا';
  }

  function clearText(){
    document.getElementById('textInput').value = '';
    document.getElementById('textResult').innerText = 'متن صاف کر دیا گیا';
  }

  function romanToUrdu(){
    // نوٹ: یہ آسان مثال ہے، اصل میں بہتر رومن اردو → اردو کنورٹر الگ ڈیٹا سے بنائیں گے
    let text = document.getElementById('textInput').value;
    text = text.replace(/salam/gi, 'سلام').replace(/khuda/gi, 'خدا');
    document.getElementById('textInput').value = text;
    document.getElementById('textResult').innerText = 'رومن اردو سے اردو میں تبدیلی مکمل';
  }

  function urduToRoman(){
    let text = document.getElementById('textInput').value;
    text = text.replace(/سلام/gi, 'salam').replace(/خدا/gi, 'khuda');
    document.getElementById('textInput').value = text;
    document.getElementById('textResult').innerText = 'اردو سے رومن اردو میں تبدیلی مکمل';
  }

  // ─────────────── ٹول 2 کرنسی کنورٹر ───────────────
  const rates = {
    PKR: { USD: 0.0036, AED: 0.013, INR: 0.30 },
    USD: { PKR: 278, AED: 3.67, INR: 83 },
    AED: { PKR: 75.7, USD: 0.27, INR: 22.5 },
    INR: { PKR: 3.3, USD: 0.012, AED: 0.044 }
  };

  function convertCurrency(){
    let amount = parseFloat(document.getElementById('amount').value);
    let from = document.getElementById('fromCurrency').value;
    let to = document.getElementById('toCurrency').value;
    if(isNaN(amount)){
      document.getElementById('currencyResult').innerText = 'براہ کرم رقم درج کریں';
      return;
    }
    if(from === to){
      document.getElementById('currencyResult').innerText = 'کرنسی ایک جیسی ہے';
      return;
    }
    let result = amount * rates[from][to];
    document.getElementById('currencyResult').innerText = amount + ' ' + from + ' = ' + result.toFixed(2) + ' ' + to;
  }

  // ─────────────── ٹول 3 PDF مرج / سپلٹ ───────────────
  function mergePDFs(){
    document.getElementById('pdfResult').innerText = '📌 PDF مرج فنکشن بعد میں ایڈوانس لائبریری سے جوڑیں گے';
  }

  function splitPDF(){
    document.getElementById('pdfResult').innerText = '📌 PDF سپلٹ فنکشن بعد میں ایڈوانس لائبریری سے جوڑیں گے';
  }
</script>
